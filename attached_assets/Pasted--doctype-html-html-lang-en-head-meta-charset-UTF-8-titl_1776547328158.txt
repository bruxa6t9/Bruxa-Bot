    <script>
      async function loadUsers() {
        const res = await fetch('/api/users');
        const users = await res.json();
        const container = document.getElementById('usersList');

        // Update stats
        document.getElementById('totalUsers').innerText = users.length;
        document.getElementById('onlineUsers').innerText = users.filter(
          (u) => u.status === 'online'
        ).length;
        document.getElementById('offlineUsers').innerText = users.filter(
          (u) => u.status === 'offline'
        ).length;

        container.innerHTML = '';

        if (users.length === 0) {
          container.innerHTML = `<div class="no-users">No users yet. Create one above!</div>`;
          return;
        }

        users.forEach((user) => {
          const isOnline = user.status === 'online';
          const lastSeen = user.lastSeen
            ? new Date(user.lastSeen).toLocaleString()
            : 'Never';
          const adminUIDs = user.adminBot?.join(', ') || 'N/A';

          container.innerHTML += `
          <div class="user-card">
            <h3>
              <span class="status ${isOnline ? 'online' : 'offline'}"></span>
              ${user.nickNameBot || 'Not Registered Yet'}
            </h3>
            <span class="badge ${isOnline ? 'badge-online' : 'badge-offline'}">
              ${isOnline ? '🟢 Online' : '⚫ Offline'}
            </span>
            <div class="info">Prefix: <span>${user.prefix || 'N/A'}</span></div>
            <div class="info">Language: <span>${user.language || 'N/A'}</span></div>
            <div class="info">Timezone: <span>${user.timeZone || 'N/A'}</span></div>
            <div class="info">API Key: <span>${user.apiKey}</span></div>
            <div class="info">Used: <span>${user.usedRequests}</span> | Remaining: <span>${user.remainingRequests}</span></div>
            <div class="info">Last Seen: <span>${lastSeen}</span></div>
            <div class="info">Admin UIDs:</div>
            <div class="admin-uids">${adminUIDs}</div>
            <br>
            <button class="btn-delete" onclick="deleteUser('${user.apiKey}')">🗑 Delete</button>
          </div>
        `;
        });
      }

      async function createUser() {
        const apiKey = document.getElementById('apiKey').value.trim();
        const msg = document.getElementById('msg');

        if (!apiKey) {
          msg.className = 'error';
          msg.innerText = 'API key is required!';
          return;
        }

        const res = await fetch('/api/create-user', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ apiKey })
        });

        const data = await res.json();
        if (res.ok) {
          msg.className = 'msg';
          msg.innerText = '✅ User created successfully!';
          document.getElementById('apiKey').value = '';
          loadUsers();
        } else {
          msg.className = 'error';
          msg.innerText = '❌ ' + data.error;
        }
      }

      async function deleteUser(apiKey) {
        if (!confirm('Delete this user?')) return;
        await fetch(`/api/users/${apiKey}`, { method: 'DELETE' });
        loadUsers();
      }

      // Auto refresh every 30 seconds
      loadUsers();
      setInterval(loadUsers, 30000);
